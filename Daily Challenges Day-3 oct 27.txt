Q1.Write a c program to addition, subtraction, multiplication, division, modulo division of two numbers.

#include<stdio.h>
int main(){
    int a,b;
    scanf("%d%d",&a,&b);
    printf("%d\n",a+b);
    printf("%d\n",a-b);
    printf("%d\n",a*b);
    printf("%d\n",a/b); 
    printf("%d\n",a%b);
    return 0;
}
